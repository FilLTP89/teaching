\newcommand{\testd}[1]%
  {\tiny{#1}\scriptsize{#1}\footnotesize{#1}\small\normalsize{#1}%
   \large{#1}\Large{#1}\LARGE{#1}\huge{#1}\Huge{#1}\normalsize{}%
  }%

  {#1_{#1_{#1_{#1}}}^{#1^{#1^{#1}}}}%

  {\mathrm{#1}\mathbf{#1}\mathsf{#1}\mathit{#1}\mathtt{#1}\mathcal{#1}}%
